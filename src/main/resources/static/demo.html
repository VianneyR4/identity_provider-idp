Custom Integration

                        
Full control with JavaScript API:

                        

const widget = new IdPWidget({
  containerId: 'idp-widget',
  apiBaseUrl: 'https://your-idp.com/api',
  clientId: 'your-client-id',
  onSuccess: (user, tokens) => {
    // Handle successful authentication
    console.log('User logged in:', user);
    // Store tokens, redirect, etc.
    localStorage.setItem('accessToken', tokens.accessToken);
    localStorage.setItem('refreshToken', tokens.refreshToken);
  },
  onError: (error) => {
    // Handle authentication errors
    console.error('Auth error:', error);
    // Show user-friendly error message
    alert('Login failed: ' + error.message);
  }
});

// Show specific views
widget.showView('login');
widget.showView('register');
                        

                    

                    
                    

                        
React Integration

                        
Easy integration with React applications:

                        

import { useEffect, useState, useRef } from 'react';

function AuthWidget() {
  const [user, setUser] = useState(null);
  const [tokens, setTokens] = useState(null);
  const widgetRef = useRef(null);
  const containerRef = useRef(null);
  
  useEffect(() => {
    if (containerRef.current && !widgetRef.current) {
      widgetRef.current = new IdPWidget({
        containerId: containerRef.current.id,
        apiBaseUrl: process.env.REACT_APP_IDP_URL,
        clientId: process.env.REACT_APP_CLIENT_ID,
        onSuccess: (user, tokens) => {
          // Update React state
          setUser(user);
          setTokens(tokens);
        },
        onError: (error) => {
          console.error('Auth error:', error);
        }
      });
    }
  }, []);
  
  return (
    <div>
      <div ref={containerRef} id="react-idp-widget"></div>
      {user && <p>Welcome, {user.email}!</p>}
    </div>
  );
}
                        

                    

                    
                    

                        
Vue.js Integration

                        
Seamless Vue.js integration:

                        

<template>
  <div>
    <div ref="widgetContainer" id="vue-idp-widget"></div>
    <p v-if="user">Welcome, {{ user.email }}!</p>
  </div>
</template>

<script>
export default {
  data() {
    return {
      widget: null,
      user: null,
      tokens: null
    };
  },
  mounted() {
    this.widget = new IdPWidget({
      containerId: 'vue-idp-widget',
      apiBaseUrl: process.env.VUE_APP_IDP_URL,
      clientId: process.env.VUE_APP_CLIENT_ID,
      onSuccess: (user, tokens) => {
        this.user = user;
        this.tokens = tokens;
        this.$emit('authenticated', { user, tokens });
      },
      onError: (error) => {
        console.error('Auth error:', error);
      }
    });
  },
  methods: {
    showLogin() {
      this.widget.showView('login');
    },
    showRegister() {
      this.widget.showView('register');
    }
  }
};
</script>
